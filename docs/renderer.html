    Create faces of MMD model. Due to the performance reason, we don't render all of faces. This is adjusted by the value interval.

      createFaces: ->
    interval = 10
    for face in @model.faces by interval
      geometry = new THREE.Geometry()
      geometry.vertices = [
        new THREE.Vector3(
          @model.vertices[face[0]].position.x,
          @model.vertices[face[0]].position.y,
          @model.vertices[face[0]].position.z
        ),
        new THREE.Vector3(
          @model.vertices[face[1]].position.x,
          @model.vertices[face[1]].position.y,
          @model.vertices[face[1]].position.z
        ),
        new THREE.Vector3(
          @model.vertices[face[2]].position.x,
          @model.vertices[face[2]].position.y,
          @model.vertices[face[2]].position.z
        ),
      ]
      geometry.faces.push(new THREE.Face3(0, 1, 2))
      geometry.computeFaceNormals()
      mesh = new THREE.Mesh(geometry, new THREE.MeshNormalMaterial())
      mesh

  createRigids: ->
    interval          = 1
    for rigid in @model.rigids by interval
      mesh = new THREE.Mesh(
        new THREE.SphereGeometry(0.5),
        new THREE.MeshNormalMaterial()
      )
      mesh.position.x = rigid.position.x
      mesh.position.y = rigid.position.y
      mesh.position.z = rigid.position.z
      @scene.add(mesh)

  createBones: ->
    interval = 1
    for bone in @model.bones by interval
      geometry        = new THREE.SphereGeometry(0.1)
      material        = new THREE.MeshNormalMaterial(wireFrame: true)
      mesh            = new THREE.Mesh(geometry, material)
      mesh.position.x = bone.position.x
      mesh.position.y = bone.position.y
      mesh.position.z = bone.position.z
      @scene.add(mesh)

      if bone.destination != -1
        material = new THREE.LineBasicMaterial(color: 0xff0000)
        geometry = new THREE.Geometry()
        geometry.vertices.push(
          new THREE.Vector3(
            bone.position.x,
            bone.position.y,
            bone.position.z
          )
        )
        destination = switch bone.flags.specifiedByIndex
          when true then @model.bones[bone.destination]
          when false then bone.destination
        geometry.vertices.push(
          new THREE.Vector3(
            destination.x,
            destination.y,
            destination.z
          )
        )
        mesh            = new THREE.Line(geometry, material)
        mesh.position.x = bone.position.x
        mesh.position.y = bone.position.y
        mesh.position.z = bone.position.z
        @scene.add(mesh)

  addPlane: ->
    geometry = new THREE.PlaneGeometry(100, 100, 10, 10)
    material = new THREE.MeshBasicMaterial(wireFrame: true, color: 0xff0000)
    mesh     = new THREE.Mesh(geometry, material)
    console.log(mesh)
    @scene.add(mesh)
